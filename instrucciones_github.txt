# Generar clave SSH RSA de 4096 bits (presiona Enter para todo por defecto)
ssh-keygen -t rsa -b 4096 -C "tu-email@example.com"

# Iniciar el agente SSH
eval "$(ssh-agent -s)"

# Añadir la clave privada recién creada al agente SSH
ssh-add ~/.ssh/id_rsa

# Mostrar la clave pública para copiar (no la privada)
cat ~/.ssh/id_rsa.pub

# --- Ahora copia la salida del comando anterior y pégala en GitHub en:
# Settings > SSH and GPG keys > New SSH key
# Dale un título y guarda.

# Cambiar la URL remota para usar SSH en vez de HTTPS
git remote set-url origin git@github.com:danieltajamar2025/bbdd_1.git

# Probar conexión SSH con GitHub
ssh -T git@github.com

# Finalmente, hacer push al repositorio
git push -u origin main
